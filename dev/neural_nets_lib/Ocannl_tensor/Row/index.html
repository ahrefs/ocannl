type constraint_ =
  1. | Dim_eq of {
    1. d1 : dim;
    2. d2 : dim;
    }
  2. | Row_eq of {
    1. r1 : t;
    2. r2 : t;
    }
  3. | Dim_ineq of {
    1. cur : dim;
    2. subr : dim;
    }
  4. | Row_ineq of {
    1. cur : t;
    2. subr : t;
    }
  5. | Dim_constr of {
    1. d : dim;
    2. constr : dim_constraint;
    }
  6. | Rows_constr of {
    1. r : t Base.list;
    2. constr : row_constraint;
    }
    (*

    The constraint applies to the concatenation of the rows. Note: broadcasting does not affect the constraint (i.e. there is no "subtyping", it resembles Row_eq).

    *)
  7. | Terminal_dim of dim
  8. | Terminal_row of t
    (*

    A row of the shape of a terminal tensor (i.e. a tensor that does not have sub-tensors).

    *)
  9. | Shape_row of t
    (*

    A row of a shape of interest.

    *)
